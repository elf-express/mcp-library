# SqlSugar — 使用倉儲（Repository Pattern）筆記

[//]: # (sqlsugar-source)
> 📖 官方文件:[果糖網 SqlSugar 5x 文件](https://www.donet5.com/Home/Doc?typeId=1228)

## 功能說明

倉儲模式讓資料存取邏輯更規範、可重用，並與業務邏輯清楚分離。
SqlSugar 的 `SimpleClient<T>` 提供完整的基礎 CRUD 方法，繼承後即可使用。

---

## 一、建立通用倉儲

```csharp
// 通用倉儲基底類（整個專案共用，不包含業務邏輯）
public class Repository<T> : SimpleClient<T> where T : class, new()
{
    public Repository(ISqlSugarClient db)
    {
        base.Context = db;  // 注入 db 實例
    }

    // 擴展方法：自帶方法不夠用時在這裡新增
    public List<T> GetByCondition(string json)
    {
        // base.Context 可拿到完整的 SqlSugarClient
        return base.Context.Queryable<T>().ToList();
    }
}
```

## 二、倉儲內建方法速查

```csharp
// ===== 查詢 =====
T      entity = base.GetById(1);                         // 主鍵查詢
List<T> all   = base.GetList();                          // 查詢所有
List<T> cond  = base.GetList(it => it.Id == 1);         // 條件查詢
T      single = base.GetSingle(it => it.Id == 1);       // 單筆（結果集不能超過1，否則報錯）
T      first  = base.GetFirst(it => it.Status == 1);    // 第一筆

// 分頁查詢
var p = new PageModel { PageIndex = 1, PageSize = 20 };
List<T> page1 = base.GetPageList(it => it.Name == "Jack", p);
List<T> page2 = base.GetPageList(it => it.Name == "Jack", p, it => it.Id, OrderByType.Desc);
Console.WriteLine(p.PageCount);  // 總頁數
Console.WriteLine(p.TotalCount); // 總筆數

// ConditionalModel 條件分頁
var conditions = new List<IConditionalModel>
{
    new ConditionalModel { FieldName = "Status", ConditionalType = ConditionalType.Equal, FieldValue = "1" }
};
List<T> page3 = base.GetPageList(conditions, p, it => it.Id, OrderByType.Desc);

// 轉 Queryable（複雜查詢）
List<T> list = base.AsQueryable().Where(it => it.Id > 0).OrderBy(it => it.Id).ToList();

// ===== 插入 =====
base.Insert(entity);                        // 單筆插入
base.InsertRange(list);                     // 批量插入
int  id          = base.InsertReturnIdentity(entity);      // 插入返回自增 ID
long snowflakeId = base.InsertReturnSnowflakeId(entity);   // 插入返回雪花 ID
base.AsInsertable(entity).IgnoreColumns(it => it.CreateTime).ExecuteCommand();  // 複雜插入

// ===== 刪除 =====
base.Delete(entity);                        // 實體刪除（需有主鍵）
base.Delete(list);                          // 集合刪除（需有主鍵）
base.DeleteById(1);                         // 主鍵刪除
base.DeleteByIds(new object[] { 1, 2, 3 }); // 多主鍵刪除
// int[] 轉 object[]：ids.Cast<object>().ToArray()
base.Delete(it => it.Id == 1);             // 表達式刪除
base.AsDeleteable().Where(it => it.Status == 0).ExecuteCommand();  // 複雜刪除

// ===== 更新 =====
base.Update(entity);                        // 實體更新（全欄位）
base.UpdateRange(list);                     // 批量更新
base.Update(it => new Order { Status = 2 }, it => it.Id == 1);  // 表達式更新（只更新 Status）
base.UpdateSetColumnsTrue(it => new Order { Status = 2 }, it => it.Id == 1);  // 更新 + 觸發 AOP 事件
base.AsUpdateable(entity).UpdateColumns(it => new { it.Name, it.Status }).ExecuteCommand();  // 複雜更新

// ===== 高階操作 =====
ISqlSugarClient db     = base.Context;       // 取得完整 db 物件
ISqlSugarClient fullDb = base.AsSugarClient; // 取得完整 db（另一種寫法）
ITenant         tenant = base.AsTenant;      // 多庫操作

// 子 db 取得主 db
ISqlSugarClient rootDb = base.Context.Root;
ITenant rootTenant     = base.Context.Root.AsTenant();
```

## 三、IOC 注入使用倉儲（推薦）

### 3.1 注入設定（Program.cs）

```csharp
// 注入 SqlSugar（以 SqlSugarScope 單例為例）
builder.Services.AddSingleton<ISqlSugarClient>(new SqlSugarScope(new ConnectionConfig
{
    DbType                = DbType.SqlServer,
    ConnectionString      = configuration.GetConnectionString("Default"),
    IsAutoCloseConnection = true,
    InitKeyType           = InitKeyType.Attribute
}));

// 注入泛型倉儲（Scoped，每個請求一個實例）
builder.Services.AddScoped(typeof(Repository<>));

// 或注入具體倉儲
builder.Services.AddScoped<OrderRepository>();
builder.Services.AddScoped<CustomerRepository>();
```

### 3.2 Controller / Service 使用

```csharp
// 注入多個倉儲（推薦，避免切換倉儲）
[ApiController]
[Route("api/[controller]")]
public class OrderController : ControllerBase
{
    private readonly Repository<Order>    _orderRepo;
    private readonly Repository<Customer> _customerRepo;
    private readonly Repository<OrderItem> _itemRepo;

    public OrderController(
        Repository<Order>    orderRepo,
        Repository<Customer> customerRepo,
        Repository<OrderItem> itemRepo)
    {
        _orderRepo    = orderRepo;
        _customerRepo = customerRepo;
        _itemRepo     = itemRepo;
    }

    [HttpGet("{id}")]
    public async Task<IActionResult> Get(int id)
    {
        var order    = await _orderRepo.GetByIdAsync(id);
        var customer = await _customerRepo.GetFirstAsync(it => it.Id == order.CustomerId);
        var items    = await _itemRepo.GetListAsync(it => it.OrderId == id);

        return Ok(new { order, customer, items });
    }

    [HttpPost]
    public async Task<IActionResult> Create([FromBody] CreateOrderDto dto)
    {
        // 跨倉儲事務
        try
        {
            _orderRepo.AsTenant().BeginTran();

            var orderId = await _orderRepo.InsertReturnIdentityAsync(dto.Adapt<Order>());

            var items = dto.Items.Select(i => new OrderItem
            {
                OrderId = orderId,
                Amount  = i.Amount
            }).ToList();
            await _itemRepo.InsertRangeAsync(items);

            _orderRepo.AsTenant().CommitTran();
            return Ok(orderId);
        }
        catch
        {
            _orderRepo.AsTenant().RollbackTran();
            throw;
        }
    }
}
```

## 四、不用 IOC 直接建立倉儲

```csharp
// Repository 改為無參數建構子
public class Repository<T> : SimpleClient<T> where T : class, new()
{
    public Repository()
    {
        base.Context = DbHelper.GetDbInstance();  // 靜態方法取得 db
    }
}

// Service 中使用（=> 表示用到時才 new，避免過早實例化）
public class OrderService : IDynamicApiController, ITransient
{
    private Repository<Order>     _orderRepo     => new Repository<Order>();
    private Repository<OrderItem> _itemRepo      => new Repository<OrderItem>();
    private ISqlSugarClient       _db            => _orderRepo.Context;

    public async Task<List<Order>> GetListAsync()
    {
        return await _orderRepo.GetListAsync();
    }
}
```

## 五、取得外部倉儲（切換倉儲）

```csharp
// 在 Order 倉儲中操作 OrderItem
var itemRepo = _orderRepo.ChangeRepository<Repository<OrderItem>>();
await itemRepo.InsertRangeAsync(items);

// 注意：ChangeRepository 需要 Repository 建構子支援可空參數
public Repository(ISqlSugarClient db = null)  // 改為可空
{
    base.Context = db;
}

// 簡易切換（只能用 ORM 內建方法，無法用擴展方法）
var itemClient = _orderRepo.Change<OrderItem>();
```

## 八、搭配 Furion 框架（AIRSET 實際架構）

```csharp
// 倉儲定義
public class Repository<T> : SimpleClient<T> where T : class, new()
{
    public Repository(ISqlSugarClient db)
    {
        base.Context = db;
    }
}

// Furion Program.cs 注入
builder.Services.AddSingleton<ISqlSugarClient>(provider =>
    new SqlSugarScope(new ConnectionConfig { ... }));
builder.Services.AddScoped(typeof(Repository<>));

// Service 繼承 ITransient（Furion 自動注冊）
public class OrderService : ITransient
{
    private readonly Repository<Order>     _orderRepo;
    private readonly Repository<OrderItem> _itemRepo;

    public OrderService(Repository<Order> orderRepo, Repository<OrderItem> itemRepo)
    {
        _orderRepo = orderRepo;
        _itemRepo  = itemRepo;
    }

    // 使用倉儲內建方法
    public async Task<PagedList<Order>> GetPageAsync(int page, int size)
    {
        var p = new PageModel { PageIndex = page, PageSize = size };
        var list = await _orderRepo.GetPageListAsync(it => it.IsDeleted == false, p,
            it => it.CreateTime, OrderByType.Desc);
        return new PagedList<Order>(list, p.TotalCount, page, size);
    }

    // 使用 Context 做複雜查詢
    public async Task<List<OrderDetailDto>> GetDetailAsync(int orderId)
    {
        return await _orderRepo.Context
            .Queryable<Order, OrderItem, Customer>(
                (o, i, c) => new JoinQueryInfos(
                    JoinType.Left, o.Id == i.OrderId,
                    JoinType.Left, o.CustomerId == c.Id))
            .Where((o, i, c) => o.Id == orderId)
            .Select((o, i, c) => new OrderDetailDto
            {
                OrderNo      = o.OrderNo,
                CustomerName = c.Name,
                ItemAmount   = i.Amount
            })
            .ToListAsync();
    }
}
```

## 注意事項

- Repository 不能是**單例**（Singleton），應為 Scoped 或 Transient
- `ChangeRepository` 需要 Repository 建構子支援可空參數 `(ISqlSugarClient db = null)`
- 多個倉儲注入（多個 Repository\<T\>）比 ChangeRepository 更直觀，推薦
- 複雜查詢用 `base.Context.Queryable<T>()` 取得完整 db 操作
- 多租戶事務寫法不同，見倉儲+多租戶文件（typeId=2405）
